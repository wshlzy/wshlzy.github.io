2. (单选题, 4.0 分) Choose the following codes' output:

						
						

	public class ControlSwitch {
public static void main(String[] args) {
int day = 3;
String dayString;

switch(day) {
	case 1:
		dayString = "Monday";
		break;
	case 2:
		dayString = "Tuesday";
		break;
	case 3:
		dayString = "Wednesday";
		break;
	default:
		dayString = "Unknown";
		}

System.out.println(dayString);
	}
}
